const submitQuery = (e: React.FormEvent<HTMLFormElement>) => {
  e.preventDefault(); // Prevent default form submission behavior

  const form = e.target as HTMLFormElement;
  const query = (form.elements.namedItem("search") as HTMLInputElement)?.value;

  if (!query) return;

  // Update the URL
  navigate(`/${BASE_URL}/kbs/${id}/search?q=${encodeURIComponent(query)}`, {
    replace: false, // Ensures a new entry is added to the browser's history stack
  });

  // Update the search query state
  setSearchQuery(query);
};
