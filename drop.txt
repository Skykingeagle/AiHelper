<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Announcement Banner Example</title>
    <style>
        /* No need for a static banner; we will create it dynamically */
    </style>
</head>
<body>
    <!-- Your Jira navigation bar and other content -->

    <script>
        function setCookie(name, value, days) {
            const expires = new Date(Date.now() + days * 864e5).toUTCString();
            document.cookie = name + '=' + encodeURIComponent(value) + '; expires=' + expires + '; path=/';
        }

        function getCookie(name) {
            const cookies = document.cookie.split('; ').reduce((r, v) => {
                const parts = v.split('=');
                return parts[0] === name ? decodeURIComponent(parts[1]) : r;
            }, '');
            return cookies;
        }

        function closeBanner() {
            setCookie('bannerClosed', 'true', 180); // Set cookie for 180 days
            const banner = document.getElementById('announcement-banner');
            if (banner) {
                banner.remove(); // Remove the banner when closed
            }
        }

        function showBanner() {
            const banner = document.createElement('div'); // Create a new div for the banner
            banner.id = 'announcement-banner'; // Set the ID for the banner
            banner.innerHTML = `
                <center>
                    <b>Do you have suggestions?</b>
                    <span style="float: right;">
                        <b style="cursor: pointer; padding-left: 10px;" onclick="closeBanner()">&#10006;</b>
                    </span>
                </center>
            `;
            document.body.insertAdjacentElement('afterbegin', banner); // Insert the banner at the top of the body
        }

        function checkBanner() {
            const bannerClosed = getCookie('bannerClosed');
            if (bannerClosed !== 'true') {
                showBanner(); // Only show the banner if it hasn't been closed
            }
        }

        // Check the banner status when the page loads
        document.addEventListener('DOMContentLoaded', function () {
            checkBanner(); // Check if the banner should be shown
        });
    </script>
</body>
</html>
