<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Announcement Banner Example</title>
</head>
<body>
    <!-- Other content of your page -->

    <script>
        function setCookie(name, value, days) {
            const expires = new Date(Date.now() + days * 864e5).toUTCString();
            document.cookie = name + '=' + encodeURIComponent(value) + '; expires=' + expires + '; path=/';
        }

        function getCookie(name) {
            const cookies = document.cookie.split('; ').reduce((r, v) => {
                const parts = v.split('=');
                return parts[0] === name ? decodeURIComponent(parts[1]) : r;
            }, '');
            return cookies;
        }

        function closeBanner() {
            setCookie('bannerClosed', 'true', 180); // Set cookie for 180 days
            const banner = document.getElementById('announcement-banner');
            if (banner) {
                banner.style.display = 'none'; // Hide the banner
            }
        }

        function showBanner() {
            const banner = document.getElementById('announcement-banner'); // Get the banner element
            if (banner) {
                banner.style.display = 'block'; // Show the banner
            }
        }

        function checkBanner() {
            const bannerClosed = getCookie('bannerClosed');
            if (bannerClosed !== 'true') {
                showBanner(); // Only show the banner if it hasn't been closed
            }
        }

        // Create the banner element and set its inner HTML
        const banner = document.createElement('div'); // Create the banner element
        banner.id = 'announcement-banner'; // Set the ID
        banner.style.display = 'none'; // Initially hide the banner
        banner.innerHTML = `
            <center>
                <b>Do you have suggestions?</b>
                <span style="float: right;">
                    <b style="cursor: pointer; padding-left: 10px;" onclick="closeBanner()">&#10006;</b>
                </span>
            </center>
        `;
        
        // Insert the banner into the body, targeting an appropriate location below the navbar
        const targetElement = document.querySelector('body'); // Adjust this to target an appropriate container if necessary
        targetElement.insertAdjacentElement('afterbegin', banner); // Insert the banner into the DOM

        // Check the banner status on page load
        document.addEventListener('DOMContentLoaded', function () {
            checkBanner(); // Check if the banner should be shown
        });
    </script>
</body>
</html>
